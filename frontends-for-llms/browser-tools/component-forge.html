                convertToReact() {
                    // Simple HTML to React conversion (basic example)
                    let reactCode = `import React from 'react';\n\n`;
                    reactCode += `export function MyComponent() {\n`;
                    reactCode += `  return (\n`;
                    
                    // Convert HTML (simplified)
                    let converted = this.html
                        .replace(/class=/g, 'className=')
                        .replace(/for=/g, 'htmlFor=')
                        .replace(//g, '{/* $1 */}');
                    
                    // Indent
                    converted = converted.split('\n').map(line => `    ${line}`).join('\n');
                    reactCode += converted;
                    reactCode += `\n  );\n}\n`;
                    
                    return reactCode;
                },
                
                copyReactCode() {
                    navigator.clipboard.writeText(this.reactCode);
                    alert('React code copied to clipboard!');
                }
